// Imports
import React, { useState } from 'react'
import {StrictMode} from 'react';
import {createRoot} from 'react-dom/client';

// üëáÔ∏è IMPORTANT: use correct ID of your root element
// this is the ID of the div in your index.html file
const rootElement = document.getElementById('root');
const root = createRoot(rootElement);

// Componente Button
const Button = ({ onClick, text }) => { 
  return (
    <>
      <button onClick={onClick}>
        {text}
      </button>
    </>
  )
}

// Componente Statistics
const Statistics = ({ good, neutral, bad, all, positive }) => {
  return (
    <>
      <p>
        Good: {good}
      </p>
      <p>
          Neutral: {neutral}
      </p>
      <p>
        bad: {bad} 
      </p>
      <p>
        All: {all} 
      </p>
      <p>
        Positive: {positive} 
      </p>
    </>
  )
}
  
// Componente App
const App = () => {
  
  // Hooks
  const [good, setGood] = useState(0)
  const [neutral, setNeutral] = useState(0)
  const [bad, setBad] = useState(0)
  const [allFeedbacks, setAllFeedbacks] = useState(0)

  const increaseGood = () => {
        setGood(good + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }
  const increaseNeutral = () => {
        setNeutral(neutral + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }
  const increaseBad = () => {
        setBad(bad + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }

  return (
    <div>
      <h1>Give feedback</h1>
      <Button onClick = {increaseGood} text = 'Good' />
      <Button onClick = {increaseNeutral} text = 'Neutral' />
      <Button onClick = {increaseBad} text = 'Bad' />
      {/* Espacio para agregar m√°s botones */}
      <h1>Statistics</h1>
      <Statistics good = {good} neutral = {neutral} bad = {bad} all={allFeedbacks} positive={good / allFeedbacks}  />
      
    </div>
  )
}

// Se renderiza el componente App en el elemento de id "root" de nuestro documento.
// ReactDOM.render(<App />, document.getElementById('root'))
root.render(
  <StrictMode>
    <App />
  </StrictMode>,
);
