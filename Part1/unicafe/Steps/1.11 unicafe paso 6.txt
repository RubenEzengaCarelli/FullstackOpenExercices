// Componente StatisticLine 
const StatisticLine = ({ text, value }) => {
  return (
    <>
      <tr>
        <td>{text}</td>    
        <td>{value}</td>    
      </tr>
    </>
  )
}

// Componente Statistics
const Statistics = ({ good, neutral, bad }) => {
  const all = good + neutral + bad
  const average = (good - bad) / all
  const positive = (good / all) * 100
    
  return all > 0 ? (
    <>
      <h1>Statistics</h1>
      <table>
        <tbody>
          <StatisticLine text="Good" value ={good} />
          <StatisticLine text="Neutral" value ={neutral} />
          <StatisticLine text="Bad" value ={bad} />
          <StatisticLine text="All" value ={all} />
          <StatisticLine text="Average" value ={average} />
          <StatisticLine text="Positive" value ={positive} />
        </tbody>
      </table>
    </>
  ) : (
      <p>
        No feedback given
      </p>
    )
}
  
// Componente App
const App = () => {
  
  // Hooks
  const [good, setGood] = useState(0)
  const [neutral, setNeutral] = useState(0)
  const [bad, setBad] = useState(0)
  const [allFeedbacks, setAllFeedbacks] = useState(0)

  const increaseGood = () => {
        setGood(good + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }
  const increaseNeutral = () => {
        setNeutral(neutral + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }
  const increaseBad = () => {
        setBad(bad + 1)
        setAllFeedbacks(allFeedbacks + 1)
      }

  return (
    <div>
      <h1>Give feedback</h1>
      <Button onClick = {increaseGood} text = 'Good' />
      <Button onClick = {increaseNeutral} text = 'Neutral' />
      <Button onClick = {increaseBad} text = 'Bad' />
      {/* Espacio para agregar m√°s botones */}
      <Statistics good = {good} neutral = {neutral} bad = {bad} />
      
    </div>
  )
}

// Se renderiza el componente App en el elemento de id "root" de nuestro documento.
// ReactDOM.render(<App />, document.getElementById('root'))
root.render(
  <StrictMode>
    <App />
  </StrictMode>,
);
